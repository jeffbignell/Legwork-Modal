Do It Live! Modals

1. Already in the DOM

$('#example-1').modal({
  background: ['#000', 'none', 0.85, true],
  openNow: true
});

Example 1

2. Dynamically Added

$('<div id="example-2"></div>').css({
  'width':'400px',
  'height':'300px',
  'background-color':'#fff'
})
.modal({
  background: ['#000', 'none', 0.85, true],
  openNow: true
});

Example 2

3. Dynamically Added With Custom Placement

$('<div id="example-3"></div>').css({
  'width':'640px',
  'height':'480px',
  'background-color':'#fff'
})
.modal({
  background: ['#000', 'none', 0.85, true],
  placement: ['20px', '50%', '0px', '-320px'],
  openNow: true
});

Example 3

4. Callbacks

$('#example-4').modal({
  background: ['#000', 'none', 0.85, true],
  openNow: true,
  onOpened: function() {
    alert('opened');
  },
  onClosed: function() {
    alert('closed');
  }
});

Example 4

5. Custom Background Image

$('#example-5').modal({
  background: ['', 'lib/img/example-bg.png', 1, true],
  openNow: true
});

Example 5

6. Open Later

var $m = $('#example-6').modal({
  background: ['#000', 'none', 0.85, true]
});

setTimeout(function() {
  $m.modal('open');
}, 500);

Example 6

7. Close

$('#example-7').modal({
  background: ['#000', 'none', 0.85, true],
  openNow: true
  onOpened: function() {
    $ref = $(this);

    setTimeout(function() {
      $ref.modal('close');
    }, 1000);
  }
});

Example 7

8. Reinitialize

$('#example-8').modal({
  background: ['#000', 'none', 0.85, true],
  openNow: true
});

$('#example-8').modal({
  background: ['#f00', 'none', 0.85, true],
  placement: ['20px', '50%', '0px', '-360px'],
  openNow: true
});
